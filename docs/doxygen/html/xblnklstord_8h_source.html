269template<class xbNodeType>
270xbInt16 xbLinkListOrd<xbNodeType>::InsertKey( const xbNodeType &ntKey, const xbString &sData ){
271
272 xbLinkListNode<xbNodeType> *p = new xbLinkListNode<xbNodeType>( ntKey, sData );
273 if( p == 0 )
274 return XB_NO_MEMORY;
275
276 if( ulNodeCnt > 0 ){
277 xbLinkListNode<xbNodeType> *currPtr = llStartPtr;
278 xbLinkListNode<xbNodeType> *prevPtr = NULL;
279
280 // find location in the chain
281 while( currPtr && ntKey > currPtr->GetKey() ){
282 prevPtr = currPtr;
283 currPtr = currPtr->GetNextNode();
284 }
285 if( currPtr && ntKey == currPtr->GetKey() && bAllowDupKeys == 0 ){
286 delete p;
287 return XB_KEY_NOT_UNIQUE;
288 }
289
290 if( currPtr == NULL ){
291 // std::cout << "at the end of the chain\n";
292 llEndPtr = p;
293 prevPtr->SetNextNode( p );
294 p->SetPrevNode( prevPtr );
295
296 } else if( currPtr->GetPrevNode() == NULL ){
297 // std::cout << "at the beginning of the chain\n";
298 p->SetNextNode( llStartPtr );
299 llStartPtr->SetPrevNode( p );
300 llStartPtr = p;
301
302 } else {
303 // std::cout << "in the middle of the chain\n";
304 p->SetNextNode( currPtr );
305 p->SetPrevNode( currPtr->GetPrevNode());
306 currPtr->SetPrevNode( p );
307 prevPtr->SetNextNode( p );
308 }
309 } else {
310 // std::cout << "first addition to the chain\n";
311 llStartPtr = p;
312 llEndPtr = p;
313 }
314 ulNodeCnt++;
315 return XB_NO_ERROR;
316}

318template<class xbNodeType>
319xbInt16 xbLinkListOrd<xbNodeType>::RemoveKey( const xbNodeType &ntKey ){
320 // Remove the first instance of ntKey from the node chain
321 xbLinkListNode<xbNodeType> *currPtr = llStartPtr;
322 xbLinkListNode<xbNodeType> *prevPtr = NULL;
323
324 while( currPtr && ntKey > currPtr->GetKey() ){
325 prevPtr = currPtr;
326 currPtr = currPtr->GetNextNode();
327 }
328
329 if( currPtr && ntKey == currPtr->GetKey()){
330// ntKey = currPtr->GetKey();
331 if( prevPtr == NULL ){ // this is the first node
332 llStartPtr = currPtr->GetNextNode();
333 // next line fails
334 if( llStartPtr ){
335 llStartPtr->SetPrevNode( NULL );
336 }
337 delete currPtr;
338 ulNodeCnt--;
339 return XB_NO_ERROR;
340 } else if( currPtr->GetNextNode() == NULL ){ // this is the last node
341 llEndPtr = prevPtr;
342 prevPtr->SetNextNode( NULL );
343 delete currPtr;
344 ulNodeCnt--;
345 return XB_NO_ERROR;
346 } else {
347
348 prevPtr->SetNextNode( currPtr->GetNextNode());
349 currPtr->GetNextNode()->SetPrevNode( prevPtr );
350 delete currPtr;
351 ulNodeCnt--;
352 return XB_NO_ERROR;
353 }
354 } else {
355 return XB_NOT_FOUND;
356 }

451template<class xbNodeType>
452xbInt16 xbLinkListOrd<xbNodeType>::UpdateForKey( const xbNodeType &ntKey, const xbString &sData ){
453
454 if( ulNodeCnt == 0 )
455 return InsertKey( ntKey, sData );
456 xbLinkListNode<xbNodeType> * currPtr = llStartPtr;
457 xbLinkListNode<xbNodeType> * prevPtr = NULL;
458 while( currPtr && ntKey > currPtr->GetKey() ) {
459 prevPtr = currPtr;
460 currPtr = currPtr->GetNextNode();
461 }
462
463 if( currPtr && ntKey == currPtr->GetKey() ) {
464 xbLinkListNode<xbNodeType> *p = new xbLinkListNode<xbNodeType>( ntKey, sData );
465 if( prevPtr )
466 prevPtr->SetNextNode( p );
467 else
468 llStartPtr = p;
469 p->SetNextNode( currPtr->GetNextNode() );
470 p->SetPrevNode( currPtr->GetPrevNode() );
471 delete currPtr;
472 return XB_NO_ERROR;
473 }
474
475 return InsertKey( ntKey, sData );
